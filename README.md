$ pip install virtualenv
$ cd my_project_folder

$ virtualenv venv
$ .\venv\Scripts\activate 
$ pip install -r requirements.txt

# lancement de l'application :

$ .\venv\Scripts\activate
$ cd carbur_backend  
$ python manage.py runserver 8001
